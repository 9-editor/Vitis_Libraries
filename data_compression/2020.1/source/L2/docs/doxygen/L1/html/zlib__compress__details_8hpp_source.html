112 template <int DATAWIDTH = 512, int BURST_SIZE = 16, int MAX_BLOCK_SIZE = 64 * 1024>
113 void mm2sZlib(const ap_uint<DATAWIDTH>* in,
114  hls::stream<ap_uint<DATAWIDTH> >& outStream,
115  hls::stream<uint32_t>& outSizeStream,
116  uint32_t input_size) {
117  const int c_byteSize = 8;
118  const int c_wordSize = DATAWIDTH / c_byteSize;
119  const int block_length = MAX_BLOCK_SIZE;
120  const int no_blocks = (input_size - 1) / block_length + 1;
121  uint32_t readBlockSize = 0, rIdx = 0;
122 
123  for (uint32_t i = 0; i < no_blocks; i++) {
124  uint32_t transferSize = block_length;
125  if (readBlockSize + block_length > input_size) transferSize = input_size - readBlockSize;
126  outSizeStream << transferSize;
127  readBlockSize += transferSize;
128  uint32_t readSize = 0;
129  bool is_pending = true;
130  while (is_pending) {
131  uint32_t pendingBytes = (transferSize > readSize) ? (transferSize - readSize) : 0;
132  is_pending = (pendingBytes > 0) ? true : false;
133  if (pendingBytes) {
134  uint32_t pendingWords = (pendingBytes - 1) / c_wordSize + 1;
135  uint32_t burstSize = (pendingWords > BURST_SIZE) ? BURST_SIZE : pendingWords;
136  gmem_read:
137  for (uint32_t midx = 0; midx < burstSize; midx++) {
138  outStream << in[rIdx + midx];
139  }
140  rIdx += burstSize;
141  readSize += burstSize * c_wordSize;
142  }
143  }
144  }
145  outSizeStream << 0;
146 }

258 void zlibCompressStreamingPacker(hls::stream<ap_uint<IN_DWIDTH> >& inStoredStream,

304  uint32_t size_read = 0;
305  inFileEos.read();
306  // Stored Block Handling
307  if (i == no_blocks - 1 && !storedBlckSize.empty()) {
308  uint32_t sizeStrdBlck = storedBlckSize.read();
309  // Stored Block Headers
310  lcl_buffer.range(prodLbuf + IN_DWIDTH - 1, prodLbuf) = 0x0;
311  lbuf_idx++;
312  prodLbuf = lbuf_idx * 8;
313  lcl_buffer.range(prodLbuf + OUT_DWIDTH - 1, prodLbuf) = sizeStrdBlck;
314  outStream << lcl_buffer;
315  outStreamEos << 0;
316  lcl_buffer >>= OUT_DWIDTH;
317  lcl_buffer.range(prodLbuf + OUT_DWIDTH - 1, prodLbuf) = ~sizeStrdBlck;
318  outStream << lcl_buffer;
319  outStreamEos << 0;
320  lcl_buffer >>= OUT_DWIDTH;
321 
322  for (uint32_t sz = 0; sz < sizeStrdBlck; sz++) {
323  // Check for PARALLEL BYTE data and write to output stream
324  if (lbuf_idx >= c_parallelByte) {
325  outStream << lcl_buffer.range(OUT_DWIDTH - 1, 0);
326  outStreamEos << 0;
327  lcl_buffer >>= OUT_DWIDTH;
328  lbuf_idx -= c_parallelByte;
329  }
330  lcl_buffer.range((lbuf_idx * 8) + IN_DWIDTH - 1, lbuf_idx * 8) = inStoredStream.read();
331  lbuf_idx++;
332  }
333  size += sizeStrdBlck;
334  for (bool eos = inStreamEos.read(); eos == false; eos = inStreamEos.read()) {
335  inStream.read();
336  }
337  inStream.read();
338  inputCmpBlockSizeStream.read();
339  continue;
340  }
341 
342  loop_aligned:
343  for (bool eos = inStreamEos.read(); eos == 0; eos = inStreamEos.read()) {
344 #pragma HLS PIPELINE II = 1
345  // Reading Input Data
346  lcl_buffer.range(prodLbuf + OUT_DWIDTH - 1, prodLbuf) = inStream.read();
347 
348  // Writing output into memory
349  outStream << lcl_buffer.range(OUT_DWIDTH - 1, 0);
350  outStreamEos << 0;
351  // Shifting by Global datawidth
352  lcl_buffer >>= OUT_DWIDTH;
353  size_read += c_parallelByte;
354  }
355 
356  uint32_t blockSize = inputCmpBlockSizeStream.read();
357  size += blockSize;
358 
359  // Calculation for byte processing
360  uint32_t leftBytes = blockSize - size_read;
361 
362  // Left bytes from each block
363  lcl_buffer.range(prodLbuf + OUT_DWIDTH - 1, prodLbuf) = inStream.read();
364  lbuf_idx += leftBytes;
365 
366  // Check for PARALLEL BYTE data and write to output stream
367  if (lbuf_idx >= c_parallelByte) {
368  outStream << lcl_buffer.range(OUT_DWIDTH - 1, 0);
369  outStreamEos << 0;
370  lcl_buffer >>= OUT_DWIDTH;
371  lbuf_idx -= c_parallelByte;
372  }
373  }
374 
375  // Left Over Data Handling
376  if (lbuf_idx && onlyOnce) {
377  if (lbuf_idx < c_parallelByte) {
378  lcl_buffer.range(OUT_DWIDTH - 1, (lbuf_idx * 8)) = 1;
379  onlyOnce = false;
380  }
381  outStream << lcl_buffer.range(OUT_DWIDTH - 1, 0);
382  lcl_buffer >>= OUT_DWIDTH;
383  outStreamEos << 0;
384  lbuf_idx = 0;
385  }
386 
387  // Last Block Processing
388  if (lbuf_idx == 0 && onlyOnce) {
389  lcl_buffer.range(IN_DWIDTH - 1, 0) = 1;
390  lbuf_idx++;
391  onlyOnce = false;
392  }
393 
394  prodLbuf = lbuf_idx * 8;
395  lcl_buffer.range(prodLbuf + OUT_DWIDTH - 1, prodLbuf) = 0x0;
396  outStream << lcl_buffer;
397  outStreamEos << 0;
398  lcl_buffer >>= OUT_DWIDTH;
399  lcl_buffer.range(prodLbuf + OUT_DWIDTH - 1, prodLbuf) = 0xFFFF;
400  outStream << lcl_buffer;
401  outStreamEos << 0;
402  lcl_buffer >>= OUT_DWIDTH;
403 
404  // CRC Processing
405  ap_uint<32> checksum = checksumStream.read();
406  lcl_buffer.range(prodLbuf + OUT_DWIDTH - 1, prodLbuf) = checksum;
407  outStream << lcl_buffer;
408  outStreamEos << 0;
409  checksum >>= OUT_DWIDTH;
410  lcl_buffer >>= OUT_DWIDTH;
411  lcl_buffer.range(prodLbuf + OUT_DWIDTH - 1, prodLbuf) = checksum;
412  outStream << lcl_buffer;
413  outStreamEos << 0;
414  lcl_buffer >>= OUT_DWIDTH;
415 
416  // Input Size for Gzip
417  if (STRATEGY == 0) {
418  lcl_buffer.range(prodLbuf + OUT_DWIDTH - 1, prodLbuf) = inSize;
419  outStream << lcl_buffer;
420  outStreamEos << 0;
421  inSize >>= OUT_DWIDTH;

432  outStreamEos << 0;
